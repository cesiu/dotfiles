% NOTE: This does not cover all the possible empty/non-empty combinations, but
%       it's enough for my foreseeable use cases.
\makeatletter
\renewcommand{\maketitle}{
    \ifx\@author\empty
        \let\theauthor\@title
        \let\thedoc\@subtitle
    \else
        \let\theauthor\@author
        \let\thedoc\@title
    \fi
    \let\thesub\@subtitle
    \let\thedate\@date

    \begin{center}
        \renewcommand{\arraystretch}{1.5}
        \begin{tabular}{c}
            \ifx\@subtitle\empty
                \usekomafont{title}{\@title} \\
            \else
                \usekomafont{title}{\@title : \@subtitle} \\
            \fi
            \midrule[\heavyrulewidth]
            \ifx\@author\empty
                \usekomafont{subtitle}{\@date} \\
            \else
                \usekomafont{subtitle}{\@author , \@date} \\
            \fi
        \end{tabular}
    \end{center}

    \thispagestyle{plain}
}
