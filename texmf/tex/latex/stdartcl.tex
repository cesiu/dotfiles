% NOTE: This does not cover all the possible empty/non-empty combinations, but
%       it's enough for my foreseeable use cases.
\makeatletter
\renewcommand{\maketitle}{
    \ifx\@author\empty
        \let\theauthor\@title
        \let\thedoc\@subtitle
    \else
        \let\theauthor\@author
        \let\thedoc\@title
    \fi
    \let\thesub\@subtitle
    \let\thedate\@date

    \begin{center}
        \renewcommand{\arraystretch}{1.5}
        \begin{tabular}{c}
            \ifx\@subtitle\empty
                \usekomafont{title}{\@title} \\
            \else
                \usekomafont{title}{\@title : \@subtitle} \\
            \fi
            \midrule[\heavyrulewidth]
            \ifx\@author\empty
                \usekomafont{subtitle}{\@date} \\
            \else
                \usekomafont{subtitle}{\@author , \@date} \\
            \fi
        \end{tabular}
    \end{center}

    \thispagestyle{plain}
}
\makeatother

% TODO: Do these need to be vboxes? Minipages result in runaway arguments when
%       combined with comments inside of a macro/environment def'n (and they
%       ruin the spaces anyways), but vboxes might be overkill (and they reset
%       the indentation too).
\newenvironment{solution}[1][]{
    \ifthenelse{\equal{#1}{}}{}{\vbox to #1}\bgroup
    \ifthenelse{\boolean{@showsolns}}{\color{red}\em}{\comment}
} {
    \ifthenelse{\boolean{@showsolns}}{}{\endcomment}
    \egroup
}

    \ifthenelse{\boolean{@showsolns}}{#1}{#2}
}

}
